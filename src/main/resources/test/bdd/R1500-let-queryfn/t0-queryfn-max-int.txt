# queryfn tests

FEATURE: queryfn-max-int
background:
  type Flight struct {id int primaryKey, wid int optional }  end
thenType: struct

---
title: max

  insert Flight {id: 55, wid: 4 }
  insert Flight {id: 56, wid: 3 }

  let x = Flight[true].wid.max()

value:a:int:4
