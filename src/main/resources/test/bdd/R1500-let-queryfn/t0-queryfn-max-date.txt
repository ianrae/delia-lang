# queryfn tests

FEATURE: queryfn-max-date
background:
  type Flight struct {id int unique, wid date optional }  end
thenType: struct

---
title: max
given:
  insert Flight {id: 55, wid: 'bet' }
  insert Flight {id: 56, wid: 'bat' }
when:
  let x = Flight[true].wid.max()
then:
value:a:string:'bet'

---
title: max empty
given:
  insert Flight {id: 55, wid: 'bet' }
  insert Flight {id: 56, wid: 'bat' }
when:
  let x = Flight[id=100].wid.max()
then:
null
